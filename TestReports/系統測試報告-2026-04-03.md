```csharp
/// <summary>
/// 變更密碼
/// </summary>
public async Task<bool> ChangePasswordAsync(int userId, ChangePasswordRequestDto request)
{
    // 1. 查詢用戶
    var user = await _context.Users.FindAsync(userId);
    if (user == null) return false;

    // 2. 驗證舊密碼
    if (!BCrypt.Net.BCrypt.Verify(request.OldPassword, user.PasswordHash))
        return false;

    // 3. 更新密碼（使用 BCrypt Hash）
    user.PasswordHash = BCrypt.Net.BCrypt.HashPassword(request.NewPassword);
    await _context.SaveChangesAsync();

    return true;
}
```

```csharp
/// <summary>
/// 變更密碼
/// </summary>
[Authorize]
[HttpPost("change-password")]
public async Task<ActionResult<ApiResponse<object>>> ChangePassword(
    [FromBody] ChangePasswordRequestDto request)
{
    // 從 JWT Token 取得 UserId
    var userIdClaim = User.FindFirst(ClaimTypes.NameIdentifier);
    if (userIdClaim == null || !int.TryParse(userIdClaim.Value, out int userId))
        return Unauthorized(ApiResponse<object>.Fail("無效的使用者驗證"));

    var result = await _authService.ChangePasswordAsync(userId, request);
    
    if (!result)
        return BadRequest(ApiResponse<object>.Fail("目前密碼錯誤"));

    return Ok(ApiResponse<object>.Ok(null, "密碼變更成功"));
}
```

### ✅ 測試14：部門排行榜配送費分攤修復 [PASS]

#### 14-A: 測試背景 [2026-04-05]
- **問題描述**: 管理員統計分析中的部門排行榜顯示 500 錯誤
- **錯誤訊息**: `The LINQ expression could not be translated`
- **根本原因**: EF Core 無法翻譯複雜的 LINQ 查詢（GroupBy + Distinct + Count）

#### 14-B: 問題分析 [CRITICAL]

**原始錯誤程式碼**：
```csharp
// StatsController.cs - GetDepartmentRanking
var stats = await _context.OrderItems
    .Include(oi => oi.User)
    .Where(oi => oi.CreatedAt >= cutoffDate)
    .GroupBy(oi => oi.User.Department)
    .Select(g => new DepartmentStatsDto
    {
        DepartmentName = g.Key,
        TotalAmount = g.Sum(oi => oi.SubTotal),
        TotalOrders = g.Count(),
        UserCount = g.Select(oi => oi.UserId).Distinct().Count(),  // ❌ 無法翻譯
        AveragePerPerson = g.Sum(oi => oi.SubTotal) / g.Select(oi => oi.UserId).Distinct().Count()
    })
    .OrderByDescending(s => s.TotalAmount)
    .ToListAsync();
```

**錯誤原因**：
- EF Core 無法將 `Distinct().Count()` 轉換為 SQL
- 嵌套的 `Select().Distinct()` 過於複雜

#### 14-C: 解決方案 [IMPLEMENTED]

**步驟 1：先載入資料到記憶體**
```csharp
// 先用 ToListAsync() 載入數據
var orderItems = await _context.OrderItems
```

```csharp
private string GenerateJwtToken(User user)
{
    var claims = new[]
    {
        new Claim(ClaimTypes.NameIdentifier, user.UserId.ToString()),
        new Claim(ClaimTypes.Email, user.Email),
        new Claim(ClaimTypes.Name, user.Name),
        new Claim(ClaimTypes.Role, user.Role),
    };

    var key = new SymmetricSecurityKey(Encoding.UTF8.GetBytes(_jwtSettings.Secret));
    var creds = new SigningCredentials(key, SecurityAlgorithms.HmacSha256);
    
    // ✅ 使用 appsettings.json 中的 ExpiryMinutes 設定
    var token = new JwtSecurityToken(
        issuer: _jwtSettings.Issuer,
        audience: _jwtSettings.Audience,
        claims: claims,
        expires: DateTime.Now.AddMinutes(_jwtSettings.ExpiryMinutes),  // 60 分鐘
        signingCredentials: creds
    );

    return new JwtSecurityTokenHandler().WriteToken(token);
}
```
